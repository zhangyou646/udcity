def matxMultiply(A, B):
    multiply = []
    if len(A[0]) != len(B):
        raise ValueError
    # B
    result = transpose(B)
    for Al in range(len(A)):
        row = []
        for Bl in range(len(result)):
            num = 0
            for Br in range(len(result[0])):
                num += A[Al][Br] * result[Bl][Br]
            row.append(num)
        multiply.append(row)
    return multiply

""" Gaussian Jordan  Ax = b.
    
        A:  
        b: 
        decPts: 4
        epsilon: 0 1.0e-16
        
     x  Ax = b 
    None Ab 
    None A 
"""
def gj_Solve(A, b, decPts=4, epsilon=1.0e-16):
    if len(A) != len(b):
        raise ValueError
    elif len(A) != len(A[0]):
        raise ValueError
    else:
        Ab = augmentMatrix(A, b)
        row, column = shape(Ab)
        for c in range(column-1):
            current_max = 0.0
            current_row = c
            for r in range(c, row):
                # 
                if abs(Ab[r][c]) > current_max:
                    current_max = abs(Ab[r][c])
                    current_row = r
            # 
            if abs(current_max) < epsilon:
                return None
            else:
                swapRows(Ab, c, current_row)
                # c1
                while abs((Ab[c][c]-1.0)) >= epsilon:
                    scaleRow(Ab, c, 1.0 / Ab[c][c])
                # c0
                for i in range(row):
                    if i != c:
                        addScaledRow(Ab,i,c,-Ab[i][c])
        matxRound(Ab, decPts)
#         print(Ab)
        result = []
        for rr in range(row):
            # 
            result.append([Ab[rr][column-1]])
        return result
